# Leer GeoJSON de estados de México
mx_states_all_data <- st_read("https://raw.githubusercontent.com/mexicovid19/Mexico-datos/master/datos/geograficos/mexico.geojson")
## Reading layer `mexico' from data source 
##   `https://raw.githubusercontent.com/mexicovid19/Mexico-datos/master/datos/geograficos/mexico.geojson' 
##   using driver `GeoJSON'
## Simple feature collection with 32 features and 13 fields
## Geometry type: MULTIPOLYGON
## Dimension:     XY
## Bounding box:  xmin: -118.4042 ymin: 14.55055 xmax: -86.73862 ymax: 32.71846
## Geodetic CRS:  WGS 84
mx_states <- mx_states_all_data %>%select(name,geometry)

# Aplicar transformación: quitar acentos, mayúsculas, reemplazar espacios con guiones
nombres_transformados <- mx_states$name %>%
  iconv(from = "UTF-8", to = "ASCII//TRANSLIT") %>%  # elimina acentos
  toupper() %>%
  gsub(" ", "_", .)  # reemplaza espacios por guiones

nombres_transformados <- ifelse(nombres_transformados == "CIUDAD_DE_MEXICO",
                                 "DISTRITO_FEDERAL",
                                 nombres_transformados)

mx_states$Estado <-nombres_transformados

3. Unir los datos geoespaciales con tus datos de β

mx_merged <- mx_states %>%
  left_join(beta_data, by = "Estado")

4. Mapas de la tasa de transmisión β

Mapa de la mediana de β

p_median <- ggplot(mx_merged) +
  geom_sf(aes(fill = Mediana), color = "white", size = 0.2) +
  scale_fill_gradientn(colours = brewer.pal(5, "PuRd"), na.value = "gray90") +
  labs(title = expression("Mediana de " * beta), fill = expression(beta)) +
  theme_minimal()
p_median

Mapa del límite inferior del intervalo de credibilidad

p_lower <- ggplot(mx_merged) +
  geom_sf(aes(fill = HDI_lower), color = "white", size = 0.2) +
  scale_fill_gradientn(colours = brewer.pal(5, "Blues"), na.value = "gray90") +
  labs(title = expression("Límite inferior del HDI de " * beta), fill = expression(beta[lower])) +
  theme_minimal()
p_lower

Mapa del límite superior del intervalo de credibilidad

p_upper <- ggplot(mx_merged) +
  geom_sf(aes(fill = HDI_upper), color = "white", size = 0.2) +
  scale_fill_gradientn(colours = brewer.pal(5, "Oranges"), na.value = "gray90") +
  labs(title = expression("Límite superior del HDI de " * beta), fill = expression(beta[upper])) +
  theme_minimal()
p_upper

5. Panel conjunto de los tres mapas

plot_grid(p_median, p_lower, p_upper, ncol = 1, labels = "AUTO")
